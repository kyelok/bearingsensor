1/*
2 * platform.h — host vs. target shims for the BWM SPU firmware.
3 *
4 * On the TI C2000 target build (TMS320F2811), TI's compiler defines
5 * __TMS320C2000__ and the codebase compiles directly against the original
6 * register-level TI headers. On the macOS / Linux host build (clang or gcc),
7 * we are running unit tests, so we macro the TI-specific keywords and
8 * pragmas to no-ops and provide stdint-backed typedefs.
9 *
10 * Application *.c files include this file once. They are not expected to
11 * branch on host vs. target themselves.
12 *
13 * IMPORTANT: On TI C28x, sizeof(char) == sizeof(short) == 1, and that 1
14 * unit is 16 bits. On host (clang, gcc, x86, arm64) char is 8 bits. Code
15 * that passes byte-counts to memcpy() vs. word-counts will behave
16 * differently on the two targets. struct-layout assertion tests at
17 * host/test/algo/test_struct_layout.c are the regression net for this.
18 */
19
20#ifndef BWM_PLATFORM_H
21#define BWM_PLATFORM_H
22
23#if defined(__TMS320C2000__)
24
25 /* Production target build under TI CGT-C2000. Everything below is what
26 * the original v6.20 firmware relied on; nothing changes for the chip. */
27 typedef unsigned int Uint16;
28 typedef int Int16;
29 typedef unsigned long Uint32;
30 typedef long Int32;
31
32#else
33
34 /* Host build: clang on macOS arm64 or gcc on Linux x86_64. */
35 #include <stdint.h>
36
37 typedef uint16_t Uint16;
38 typedef int16_t Int16;
39 typedef uint32_t Uint32;
40 typedef int32_t Int32;
41
42 /* TI-specific keywords vanish on host. */
43 #define interrupt
44 #define __interrupt
45 #define near
46 #define far
47 #define cregister
48
49 /* TI's EALLOW / EDIS guard register-write protection. No-op on host. */
50 #define EALLOW ((void)0)
51 #define EDIS ((void)0)
52
53 /* Clang ignores #pragma DATA_SECTION / CODE_SECTION with a warning;
54 * silenced project-wide by -Wno-unknown-pragmas in the Makefile. We
55 * do NOT macro them because they are part of the source tree and
56 * the warning is informational only. */
57
58#endif
59
60#endif /* BWM_PLATFORM_H */
