1/*
2 * src/vendor/mb_sum.h — Main Bearing neighbour sum (vendor extension).
3 *
4 * @vendor-extension MB-Sum (no MAN BWM spec section).
5 * Documented at: host/docs/vendor_extensions/mb_sum.md
6 *
7 * MB-Sum is the firmware's fourth alarm metric (after spec's
8 * SingleRapid, CylRapid, TwinCylRapid). It targets MAIN BEARING wear
9 * by summing across-cylinder sensor pairs at each main-bearing
10 * position.
11 *
12 * For an N-cylinder engine there are N-1 MB positions (one between
13 * each adjacent pair of cylinders). At MB position k (zero-indexed),
14 * the inputs are:
15 * - aft sensor of cylinder k+1
16 * - fore sensor of cylinder k+2
17 *
18 * The arithmetic is shape-identical to TwinCylRapid; the difference
19 * is INTENT (main bearings vs. crosshead/crankpin) and SEPARATELY
20 * TUNABLE alarm threshold (Modbus register 4624).
21 */
22
23#ifndef BWM_VENDOR_MB_SUM_H
24#define BWM_VENDOR_MB_SUM_H
25
26#include "core/types.h"
27
28/* For an N-cylinder engine, there are N-1 MB positions. */
29static inline Uint16 mb_sum_num_positions(Uint16 num_cylinders)
30{
31 return (num_cylinders > 0) ? (Uint16)(num_cylinders - 1) : 0;
32}
33
34/* Sensor IDs for MB position k:
35 * sensor_aft = aft sensor of cylinder k+1 = (k+1)*2 + 1 = 2k+3
36 * sensor_fore = fore sensor of cylinder k+2 = (k+2)*2 = 2k+4
37 * (Zero-indexed cylinders.) */
38static inline bwm_sensor_id_t mb_sum_left_sensor (bwm_mb_id_t k) { return (bwm_sensor_id_t)(2*k + 1); }
39static inline bwm_sensor_id_t mb_sum_right_sensor(bwm_mb_id_t k) { return (bwm_sensor_id_t)(2*k + 2); }
40
41/* MB-Sum value at one position. */
42float mb_sum_combine(float rw_left_aft, float rw_right_fore);
43
44/* Bitmask helper: position k is valid if both adjacent sensors are
45 * enabled and not failed/replaced, AND k is not at the bank-split
46 * position (ccsplit). Returns 1 if the position should produce an
47 * MB-Sum value, 0 if not. */
48int mb_sum_position_is_valid(int sensor_left_aft_failed,
49 int sensor_right_fore_failed,
50 int sensor_left_aft_replacing,
51 int sensor_right_fore_replacing,
52 Uint16 cc_split_position,
53 bwm_mb_id_t k);
54
55#endif /* BWM_VENDOR_MB_SUM_H */
