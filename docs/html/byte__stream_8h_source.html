1/*
2 * byte_stream.h — 8-bit byte-stream helpers for serialization paths.
3 *
4 * MOTIVATION
5 *
6 * Some firmware code paths genuinely deal in 8-bit bytes — Modbus
7 * payloads, CSV strings, SD card serialization formats, the XML output
8 * required by spec 8.6 / 8.7 §6. On a stdint host these are just byte
9 * arrays. On the TI C2000 with CHAR_BIT == 16, an "8-bit byte stream"
10 * needs explicit packing — two logical bytes per word, accessed via
11 * shift/mask.
12 *
13 * STATUS: SKELETON
14 *
15 * This header is a placeholder. The actual byte-stream abstraction is
16 * needed for:
17 *
18 * - SD card serialization (sdcard.c) — when we extract the CSV header
19 * formatter, the spec 8.7 Appendix E hit-by-hit format, and the
20 * spec 8.7 Appendix F compensation-curve format.
21 *
22 * - Modbus payload assembly (PCModBus.c) — already byte-oriented in
23 * v6.20 (it stuffs payload bytes into 16-bit Modbus registers); the
24 * existing pattern works on TI, may need adjusting on host.
25 *
26 * - XML output for the surveyor-presentation file (spec §6.4) — only
27 * needed in Phase 4; defer until then.
28 *
29 * The scaffolding will be filled in during the corresponding Phase 3
30 * extraction. Until then, this header documents the intended surface
31 * so call sites know what to migrate toward.
32 *
33 * INTENDED API (sketch)
34 *
35 * typedef struct ByteStream ByteStream;
36 *
37 * void byte_stream_init(ByteStream* s, Uint16* backing, Uint16 capacity_bytes);
38 * int byte_stream_put(ByteStream* s, Uint8 byte); // returns 0 on overflow
39 * int byte_stream_put_n(ByteStream* s, const Uint8* src, Uint16 n);
40 * Uint8 byte_stream_get(ByteStream* s, Uint16 byte_index);
41 * Uint16 byte_stream_size_bytes(const ByteStream* s);
42 * Uint16 byte_stream_size_words(const ByteStream* s); // for sizing FRAM/SD writes
43 *
44 * On TI: backing is Uint16[]; byte_stream_put packs (byte_count & 1)
45 * to choose the high or low half-word and combines via OR.
46 * On host: backing is still Uint16[] for symmetry, but each Uint16
47 * carries one byte in its low 8 bits (upper 8 bits zeroed). This
48 * trades efficiency for code symmetry; call sites needing raw byte
49 * arrays on host should use byte_stream_get_byte_array() (TBD).
50 *
51 * VENDORED FROM TI BIOS PATTERNS
52 *
53 * TI's BIOS / driverlib for SCI byte streams already does this kind of
54 * packing. When we get to Phase 3 SD card extraction, we'll either
55 * lift their __byte_peripheral access pattern or reimplement here.
56 */

67/*
68 * Helper for the most common sizing question: "how many TI words do I
69 * need to hold `byte_count` packed bytes?"
70 *
71 * On TI: ceil(byte_count / 2) — two bytes per word.
72 * On host: byte_count — one byte per byte (no packing).
73 *
74 * Defined as a macro so it works in const-context (e.g., array sizing).
75 */
76#if defined(__TMS320C2000__)
77 #define WORDS_FOR_BYTES(byte_count) ( ((byte_count) + 1) / 2 )
78#else
79 #define WORDS_FOR_BYTES(byte_count) ((byte_count) * 0 + (byte_count)) /* identity, kept symmetrical */
80#endif
81
82#endif /* BWM_BYTE_STREAM_H */
